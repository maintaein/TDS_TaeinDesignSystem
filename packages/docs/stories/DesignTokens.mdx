## 📖 사용 예시

### TypeScript에서 토큰 import

```typescript
import { primary, gray, success } from '@designsystem/core/tokens/colors';
import { fontSize, fontWeight } from '@designsystem/core/tokens/typography';
import { spacing, borderRadius } from '@designsystem/core/tokens/spacing';
import { shadows } from '@designsystem/core/tokens/shadows';
import { duration, easing } from '@designsystem/core/tokens/animation';

// 사용 예시
const buttonStyle = {
  backgroundColor: primary[600],
  color: gray[50],
  fontSize: fontSize.base,
  fontWeight: fontWeight.semibold,
  padding: `${spacing[3]} ${spacing[6]}`,
  borderRadius: borderRadius.base,
  boxShadow: shadows.md,
  transition: `all ${duration.base} ${easing.easeInOut}`,
};
```

### Vanilla Extract CSS에서 사용

```typescript
import { style } from '@vanilla-extract/css';
import { primary, gray } from '@designsystem/core/tokens/colors';
import { spacing, borderRadius } from '@designsystem/core/tokens/spacing';

export const button = style({
  backgroundColor: primary[600],
  color: gray[50],
  padding: `${spacing[3]} ${spacing[6]}`,
  borderRadius: borderRadius.base,

  ':hover': {
    backgroundColor: primary[700],
  },
});
```
